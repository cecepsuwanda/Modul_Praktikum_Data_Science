\begin{subcpmk}

\end{subcpmk}

\SesiRPS{5}{P4}{$\sim$170 menit}{CPMK-P2}

\section{Kemampuan akhir sesi}

Menghasilkan histogram/KDE, scatter atau boxplot kategori, heatmap korelasi; setiap plot memiliki judul dan label sumbu; merumuskan satu hipotesis pemodelan.

\section{Teori}

\begin{konsep}
EDA menjawab ``apa yang data ceritakan'' sebelum asumsi model. Visual yang baik memiliki \textbf{judul}, \textbf{label sumbu}, dan \textbf{skala} yang terbaca; hindari chart junk berlebihan.
\end{konsep}

\section{Sarana}

\texttt{matplotlib.pyplot}, \texttt{seaborn}, \texttt{pandas}.

\section{Contoh kode}

\begin{lstlisting}[caption=Histogram dan KDE]
import matplotlib.pyplot as plt
import seaborn as sns
# df, kolom numerik n1, n2
fig, ax = plt.subplots(1, 2, figsize=(10, 4))
sns.histplot(df["n1"], kde=True, ax=ax[0])
ax[0].set_title("Distribusi n1")
sns.scatterplot(data=df, x="n1", y="n2", ax=ax[1])
ax[1].set_title("n1 vs n2")
plt.tight_layout()
plt.show()
\end{lstlisting}

\begin{lstlisting}[caption=Boxplot per kategori]
plt.figure(figsize=(8, 4))
sns.boxplot(data=df, x="kategori", y="n1")
plt.title("Sebaran n1 per kategori")
plt.xticks(rotation=30)
plt.show()
\end{lstlisting}

\begin{lstlisting}[caption=Heatmap korelasi]
num = df.select_dtypes(include="number")
plt.figure(figsize=(7, 5))
sns.heatmap(num.corr(), annot=False, cmap="vlag", center=0)
plt.title("Korelasi antar fitur numerik")
plt.show()
\end{lstlisting}

\begin{lstlisting}[caption=Pairplot subset kecil]
cols = ["n1", "n2", "n3"]
sns.pairplot(df[cols].dropna(), corner=True)
plt.show()
\end{lstlisting}
