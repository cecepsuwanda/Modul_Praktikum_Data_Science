% Teks sebelum \begin{lstlisting} memakai judul yang sama dengan caption listing
% (\lstlistingname~\thelstlisting). \MulaiPenjelasanKode melakukan \refstepcounter{lstlisting};
% patch mencegah double step pada environment berikut.
\makeatletter
\newif\if@pjk@skip@lst@ref
\newcommand{\pjk@lst@refornot}[1]{%
  \if@pjk@skip@lst@ref
    \global\@pjk@skip@lst@reffalse
  \else
    \refstepcounter{#1}%
  \fi
}
\newcommand{\MulaiPenjelasanKode}{%
  \refstepcounter{lstlisting}%
  \global\@pjk@skip@lst@reftrue
  \par\vspace{0.75\baselineskip}%
  \noindent\textbf{\lstlistingname~\thelstlisting.} \begingroup\RaggedRight\sloppy\ignorespaces
}
\newcommand{\SelesaiPenjelasanKode}{\par\endgroup\vspace{0.75\baselineskip}}

    \ifx\lst@@caption\@empty\expandafter\lst@HRefStepCounter \else
                                \expandafter\refstepcounter
        \fi {lstlisting}}{%
    \ifx\lst@@caption\@empty\expandafter\lst@HRefStepCounter \else
                                \expandafter\pjk@lst@refornot
        \fi {lstlisting}}{}{%
  \PackageError{modul_praktikum_data_science}{Gagal menambal \string\lst@MakeCaption.}{}}
\makeatother
